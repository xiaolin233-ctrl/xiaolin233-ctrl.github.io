存档功能代码

// 游戏状态对象
const gameState = {
    score: 1250,
    level: 3,
    health: 85,
    playTime: 2730, // 秒
    items: ['sword', 'shield', 'potion'],
    settings: {
        sound: true,
        difficulty: 'medium'
    }
};

// 保存游戏函数
function saveGame() {
    // 将游戏状态转换为JSON字符串
    const data = JSON.stringify(gameState, null, 2);
    
    // 创建Blob对象
    const blob = new Blob([data], { type: 'text/plain' });
    
    // 创建下载链接
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `game_save_${new Date().toISOString().slice(0, 10)}.txt`;
    
    // 触发下载
    document.body.appendChild(a);
    a.click();
    
    // 清理
    setTimeout(() => {
        document.body.removeChild(a);
        URL.revokeObjectURL(url);
    }, 100);
    
    // 显示成功消息
    showMessage('游戏已成功保存为TXT文件！', 'save-success');
}

// 加载游戏函数
function loadGame(event) {
    const file = event.target.files[0];
    if (!file) return;
    
    const reader = new FileReader();
    reader.onload = function(e) {
        try {
            // 解析JSON数据
            const data = JSON.parse(e.target.result);
            
            // 验证数据完整性
            if (!data.score || !data.level) {
                throw new Error('存档文件格式不正确');
            }
            
            // 更新游戏状态
            Object.assign(gameState, data);
            
            // 更新UI
            updateUI();
            
            // 显示成功消息
            showMessage('游戏进度已成功加载！', 'load-success');
        } catch (err) {
            // 显示错误消息
            showMessage('加载存档失败：' + err.message, 'error');
        }
    };
    
    reader.readAsText(file);
    fileInput.value = ''; // 重置文件输入
}

// 显示状态消息
function showMessage(message, className) {
    const statusMessage = document.getElementById('statusMessage');
    statusMessage.textContent = message;
    statusMessage.className = 'status-message ' + className;
}

// 初始化事件监听
document.getElementById('saveBtn').addEventListener('click', saveGame);
document.getElementById('loadBtn').addEventListener('click', () => {
    document.getElementById('fileInput').click();
});
document.getElementById('fileInput').addEventListener('change', loadGame);

关键点说明：

  • 使用 JSON.stringify() 将游戏状态转换为字符串
  • 通过 Blob 对象创建可下载的TXT文件
  • 利用 FileReader API 读取上传的存档文件
  • 使用 JSON.parse() 解析存档数据
  • 数据完整性验证确保存档文件有效
  • 使用 Object.assign() 更新游戏状态
